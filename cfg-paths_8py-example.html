48 def print_paths(blocks, edges, target_id, vertex_id, visited=set(), path=[]):
49  """Depth-first search of a graph, printing all paths between two vertices.
50  """
51 
52  rv = 0
53  visited.add(vertex_id)
54  path.append(blocks[vertex_id].address)
55 
56  # At target, print the path
57  if vertex_id == target_id:
58  print(", ".join("0x%.8x" % s for s in path))
59  rv = 1
60 
61  # Otherwise, check all outgoing edges from this vertex
62  else:
63  for v in edges[vertex_id]:
64  if v not in visited and v not in unreachability_cache:
65  rv += print_paths(blocks, edges, target_id, v, visited, path)
66 
67  path.pop()
68  # Unmark the node so it can be visited again in other paths
69  visited.discard(vertex_id)
70  if rv == 0:
71  unreachability_cache.add(vertex_id)
72  return rv
