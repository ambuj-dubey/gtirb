126  template <typename Schema> const typename Schema::Type* getAuxData() const {
127  auto Found = this->AuxDatas.find(Schema::Name);
128 
129  if (Found == this->AuxDatas.end())
130  return nullptr;
131 
132  AuxData& AD = *(Found->second);
133 
134  assert(AD.getApiTypeId() != AuxData::UNREGISTERED_API_TYPE_ID &&
135  "Attempting to retrieve AuxData with an unregistered type.");
136 
137  assert(AD.getApiTypeId() == AuxDataImpl<Schema>::staticGetApiTypeId() &&
138  "Attempting to retrieve AuxData with incorrect type.");
139  auto& ADI = static_cast<AuxDataImpl<Schema>&>(AD);
140  return ADI.get();
141  }

261  template <
262  class MessageType,
263  class = std::enable_if_t<message_has_aux_data_container_v<MessageType>>>
264  void fromProtobuf(const MessageType& Message) {
265  this->AuxDatas.clear();
266  for (const auto& M : Message.aux_data()) {
267  std::unique_ptr<AuxData> Val;
268  std::string Key = M.first;
269 
270  // See if the name for this AuxData is registered.
271  if (const auto* ADT = lookupAuxDataType(Key)) {
272  Val = ADT->fromProtobuf(M.second);
273  }
274 
275  // If it wasn't registered or was registered with an incompatible
276  // type wrt the serialized object, unserialized as just the
277  // un-typed raw data.
278  if (!Val) {
279  Val = std::make_unique<AuxData>();
280  AuxData::fromProtobuf(*Val, M.second);
281  }
282  this->AuxDatas.insert(std::make_pair(Key, std::move(Val)));
283  }
285  }
