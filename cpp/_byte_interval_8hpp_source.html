1423  template <typename ByteIntervalType, typename T> class BytesReference {
1424  public:
1425  BytesReference(ByteIntervalType* BI_, size_t I_,
1426  boost::endian::order InputOrder_,
1427  boost::endian::order OutputOrder_)
1428  : BI(BI_), I(I_), InputOrder(InputOrder_), OutputOrder(OutputOrder_) {}
1429 
1435  operator T() const {
1436  assert(I + sizeof(T) <= BI->Size &&
1437  "read into interval's bytes out of bounds!");
1438 
1439  auto S = BI->Bytes.size();
1440 
1441  if (I >= S) {
1442  // anything this far past the end of initialized bytes is composed of
1443  // all zero bytes, so we return what T would have been interpreted as
1444  // if all bytes are zero.
1445  //
1446  // (note that you may be tempted to replace this with "return T{};",
1447  // but beware: T might be a non-scalar type whose default constructor
1448  // differs from the value returned when all bytes are re-interpeted as
1449  // zeroes. A similar argument exists for "return T{0};".)
1450  const std::array<uint8_t, sizeof(T)> Array{};
1451  return *reinterpret_cast<const T*>(Array.data());
1452  }
1453 
1454  if (I + sizeof(T) > S) {
1455  // Here, I < S < I + sizeof(T), so we need to partially fill the
1456  // initialized bytes and combine it with zeroes for the uninitialized
1457  // bytes.
1458  std::array<uint8_t, sizeof(T)> Array{};
1459  // Thanks to math, 0 < S - I < sizeof(T).
1460  std::copy_n(BI->Bytes.begin() + I, S - I, Array.begin());
1461  return boost::endian::conditional_reverse(
1462  *reinterpret_cast<const T*>(Array.data()), InputOrder, OutputOrder);
1463  }
1464 
1465  return boost::endian::conditional_reverse(
1466  *reinterpret_cast<const T*>(BI->Bytes.data() + I), InputOrder,
1467  OutputOrder);
1468  }
