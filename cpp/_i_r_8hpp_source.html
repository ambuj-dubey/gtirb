160  bool removeModule(Module* S) {
161  auto& Index = Modules.get<by_pointer>();
162  if (auto Iter = Index.find(S); Iter != Index.end()) {
163  Index.erase(Iter);
164  S->setIR(nullptr);
165  return true;
166  }
167  return false;
168  }
169 
173  Module* addModule(Module* M) {
174  if (M->getIR()) {
175  M->getIR()->removeModule(M);
176  }
177  Modules.emplace(M);
178  M->setIR(this);
179  return M;
180  }
